62int calculateBaseAngle(Arm *self, double sensor_position_distance, double sensor_reading_distance, double sensor_angle_rad) {
63 /*$d_s*/
64 double target_distance = sqrt(sq(sensor_position_distance) + sq(sensor_reading_distance) - 2 * sensor_position_distance * sensor_reading_distance * cos(sensor_angle_rad));
65 double sin_theta_O = sin(sensor_angle_rad) * sensor_reading_distance / target_distance;
66 double cos_theta_O = (sq(target_distance) + sq(sensor_position_distance) - sq(sensor_reading_distance)) / (2 * target_distance * sensor_position_distance);
67 self->base_angle = atan2(sin_theta_O, cos_theta_O);
68 self->final_targetUV = {.u=target_distance, .v=0};
69 return 0;
70}

76int calculateNextTargetUV(Arm *self) {
77 Vector2 final_targetUV = self->final_targetUV;

81 double safe_v = final_targetUV.v + GRAB_HEIGHT;

112int calculateArmAngles(Arm *self) {
113 Vector2 final_targetUV = self->final_targetUV;
114 Vector2 target = self->targetUV;
115 self->theta2 = acos((sq(target.u) + sq(target.v) - sq(L1) - sq(L2)) / (2 * L1 * L2));
116 double k1 = L1 + L2 * cos(self->theta2);
117 double k2 = L2 * sin(self->theta2);
118 self->theta1 = atan2(target.v, target.u) - atan2(k2, k1);
119 self->theta3 = 2 * PI - self->theta1 - self->theta2 - PI / 2;
120 return 0;
121}
