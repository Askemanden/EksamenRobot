148 DetectedColor categorizeColor(uint16_t r, uint16_t g, uint16_t b, uint16_t c)
149 {
150
151 if (c < 200)
152 {
153 return COLOR_BLACK;
154 }
155 if (c > 4000)
156 {
157 return COLOR_WHITE;
158 }
159
160 float rn = (float)r;
161 float gn = (float)g;
162 float bn = (float)b;
163
164 const float minDiff = 0.03;
165 const float minRatio = 1.025;
166
167 if (rn > gn * minRatio && rn > bn * minRatio && (rn - gn) > minDiff && (rn - bn) > minDiff)
168 {
169 return COLOR_RED;
170 }
171
172 if (gn > rn * minRatio && gn > bn * minRatio && (gn - rn) > minDiff && (gn - bn) > minDiff)
173 {
174 return COLOR_GREEN;
175 }
176
177 if (bn > rn * minRatio && bn > gn * minRatio && (bn - rn) > minDiff && (bn - gn) > minDiff)
178 {
179 return COLOR_BLUE;
180 }
181
182 return COLOR_UNCERTAIN;
183 }

310 int calculateFinalTarget(Arm *self, double sensor_position_distance, double sensor_reading_distance, double sensor_angle_rad)
311 {
312 /*$d_s*/
313 double target_distance = sqrt(sq(sensor_position_distance) + sq(sensor_reading_distance) - 2 * sensor_position_distance * sensor_reading_distance * cos(sensor_angle_rad));
314 double sin_theta_O = sin(sensor_angle_rad) * sensor_reading_distance / target_distance;
315 double cos_theta_O = (sq(target_distance) + sq(sensor_position_distance) - sq(sensor_reading_distance)) / (2 * target_distance * sensor_position_distance);
316 self->base_angle = atan2(sin_theta_O, cos_theta_O);
317 self->final_targetUV = {.u = target_distance, .v = 0};
318 return 0;
319 }

325 int calculateNextTargetUV(Arm *self)
326 {
327 Vector2 final_targetUV = self->final_targetUV;
328
329 Vector2 previous_target = self->targetUV;
330
331 double safe_v = final_targetUV.v + GRAB_HEIGHT;
332
333 if (abs(previous_target.u - final_targetUV.u) < 0.05 && abs(previous_target.v - final_targetUV.v) < 0.05){
334 return 1;
335 }
336
337 if (abs(previous_target.u - final_targetUV.u) < 0.01)
338 {
339 self->targetUV = vectorAdd(previous_target, {.u = 0, .v = -STEP_DIST});
340 }
341
342 else if (previous_target.v < safe_v)
343 {
344 self->targetUV = vectorAdd(previous_target, {.u = 0, .v = STEP_DIST});
345 }
346
347 else if (previous_target.u < final_targetUV.u)
348 {
349 self->targetUV = vectorAdd(previous_target, {.u = STEP_DIST, .v = 0});
350 }
351 else
352 {
353 self->targetUV = vectorAdd(previous_target, {.u = -STEP_DIST, .v = 0});
354 }
355
356 if (abs(self->targetUV.u - final_targetUV.u) < STEP_DIST)
357 {
358 self->targetUV.u = final_targetUV.u;
359 }
360 if (abs(self->targetUV.v - final_targetUV.v) < STEP_DIST)
361 {
362 self->targetUV.v = final_targetUV.v;
363 }
364
365 return 0;
366 }

372 int calculateArmAngles(Arm *self)
373 {
374 Vector2 final_targetUV = self->final_targetUV;
375 Vector2 target = self->targetUV;
376 self->theta2 = -acos((sq(target.u) + sq(target.v) - sq(L1) - sq(L2)) / (2 * L1 * L2));
377 double k1 = L1 + L2 * cos(self->theta2);
378 double k2 = L2 * sin(self->theta2);
379 self->theta1 = atan2(target.v, target.u) - atan2(k2, k1);
380 self->theta3 = 2 * PI - self->theta1 - self->theta2 - PI / 2;

457
458 void dc_motor_update(MotorUnit &m)
459 {
460 long error = m.target_position - m.current_position;
461 if (m.finished)
462 {
463 turn(m, false, 0);
464 return;
465 }
466
467 if (abs(error) <= threshold)
468 {
469 turn(m, false, 0);
470 if (m.finished_ticks >= 250)
471 {
472 m.finished = true;
473 m.finished_ticks = 0;
474 return;
475 }
476 else
477 {
478 m.finished_ticks++;
479 }
480 }
481
482 m.pid_data.integral += error;
483 m.pid_data.integral = constrain(m.pid_data.integral, -1000, 1000);
484
485 double derivative = (error - m.pid_data.last_error);
486 m.pid_data.last_error = error;
487
488 double output =
489 m.pid_data.kp * error +
490 m.pid_data.ki * m.pid_data.integral +
491 m.pid_data.kd * derivative;
492
493 int speed = abs(output);
494
495 turn(m, (output > 0), constrain(speed, (int)m.MIN_POWER, MAX_SPEED));
496 }

652float measureDistance(const UltrasonicSensor &s) {
653 digitalWrite(s.trigPin, LOW);
654 delayMicroseconds(2);
655 digitalWrite(s.trigPin, HIGH);
656 delayMicroseconds(10);
657 digitalWrite(s.trigPin, LOW);

660 if (duration == 0) return -1;
661 return (duration / 58.0) + distanceOffsetcm;

664float averageDistance(const UltrasonicSensor &s, int samples) {
665 float sum = 0;
666 int valid = 0;
667 for (int i = 0; i < samples; i++) {
668 float d = measureDistance(s);
669 if (d > 0) { sum += d; valid++; }
670 delayMicroseconds(500);
671 }
672 if (valid == 0) return -1;
673 return sum / valid;

676ScanResult performScan(const UltrasonicSensor &s, ServoScanner &sc, int startAngle, int endAngle) {
677 ScanResult result = {-1, -1, -1, -1, -1, false, -1};
678
679 int step = (endAngle >= startAngle) ? sc.stepSize : -sc.stepSize;
680 int prevAngle = startAngle;
681 bool wasDetected = false;
682 float distanceSum = 0;
683 int distanceCount = 0;
684
685 int absoluteMin = sc.minAngle - EDGE_EXTENSION;
686 int absoluteMax = sc.maxAngle + EDGE_EXTENSION;
687 bool inExtension = false;
688
689 int dynamicEnd = endAngle;
690
691 for (int angle = startAngle;
692 (step > 0) ? (angle <= dynamicEnd) : (angle >= dynamicEnd);
693 angle += step)
694 {
695 int servoAngle = constrain(angle, absoluteMin, absoluteMax);
696 sc.servo.write(servoAngle);
697 delay(30);
698
699 float dist = averageDistance(s, 3);
700 bool isDetected = (dist > 0 && dist <= detectionThresholdcm);
701
702 if (!wasDetected && isDetected) {
703 result.edgeBefore = prevAngle;
704 result.detected = true;
705 }
706
707 if (isDetected) {
708 distanceSum += dist;
709 distanceCount++;
710 }
711
712 if (wasDetected && !isDetected) {
713 result.edgeAfter = angle;
714 break;
715 }
716
717 if (isDetected) {
718 bool atNormalEnd = (step > 0) ? (angle >= endAngle) : (angle <= endAngle);
719 if (atNormalEnd && !inExtension) {
720 inExtension = true;
721 dynamicEnd = endAngle + (step * EDGE_EXTENSION);
722 dynamicEnd = constrain(dynamicEnd, absoluteMin, absoluteMax);
723 }
724 }
725
726 wasDetected = isDetected;
727 prevAngle = angle;
728 }
729
730 if (result.detected && result.edgeAfter == -1) {
731 result.edgeAfter = dynamicEnd;
732 }
733
734 if (result.detected) {
735 result.trueAngle = (result.edgeBefore + result.edgeAfter) / 2;
736 if (distanceCount > 0) result.objectDistance = distanceSum / distanceCount;
