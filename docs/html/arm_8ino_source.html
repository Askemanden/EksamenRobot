68 DetectedColor categorizeColor(uint16_t r, uint16_t g, uint16_t b, uint16_t c) {

204 int calculateBaseAngle(Arm *self, double sensor_position_distance, double sensor_reading_distance, double sensor_angle_rad) {
205 /*$d_s*/

207 double sin_theta_O = sin(sensor_angle_rad) * sensor_reading_distance / target_distance;
208 double cos_theta_O = (sq(target_distance) + sq(sensor_position_distance) - sq(sensor_reading_distance)) / (2 * target_distance * sensor_position_distance);

254 int calculateArmAngles(Arm *self) {
255 Vector2 final_targetUV = self->final_targetUV;
256 Vector2 target = self->targetUV;
257 self->theta2 = acos((sq(target.u) + sq(target.v) - sq(L1) - sq(L2)) / (2 * L1 * L2));
